# Installation

# Install library

npm install nanoutils

# Use with ES modules

To use ES modules we wrote a plugin babel-plugin-nanoutils.

# Install plugin

npm install babel-plugin-nanoutils

# Usage

# Via .babelrc:

{
  "plugins": ["nanoutils"]
}

# Via CLI

babel --plugins nanoutils script.js

# Via Node API

require("babel-core").transform("code", {
  plugins: ["nanoutils"]
})
